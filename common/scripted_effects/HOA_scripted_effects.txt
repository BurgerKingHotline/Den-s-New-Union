HOA_anger_rise_daily = {
    if = {
        limit = {
            check_variable = {
                HOA_Anger < 100
            }
        }

        hidden_effect = {
            add_to_variable = {
                var = HOA_Anger
                value = 1
            }
        }
    }
}
